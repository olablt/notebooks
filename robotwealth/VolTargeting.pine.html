 6 // @description: Functions for calculating position sizes based on volatility targets
 7 library("VolTargeting")
 8 import TradingView/ta/7
 9 
10 // @function Calculates an exponentially weighted annualised volatility
11 // @param estimationWdw Half life of exponential weighting scheme
12 // @returns Exponentially weighted annualised volatility
13 export ewmaVol(int estimationWdw) =>
14     // vol annualisation factor  
15     // scale by sqrt(365) for crypto, sqrt(252) for everything else
16     int secondsInYear = 252*24*60*60
17     if syminfo.type == "crypto"
18         secondsInYear := 365*24*60*60
19     volAnnualisationFactor = math.sqrt(secondsInYear/timeframe.in_seconds())
20 
21     // vol calcs
22     simple_return = (close[0] - close[1])/close[1]
23     ewmaVariance = ta.ema(math.pow(simple_return, 2), estimationWdw)
24     ewmaVolatility = math.sqrt(ewmaVariance) * volAnnualisationFactor
25 
26     ewmaVolatility
27 
28 // @function Calculates a volatility-targetted position size in number of units
29 // @param nominalBalance Portfolio size on which to calculate volatility targets
30 // @param targetVolContribution Amount of volatility to be assigned to this position
31 // @param forecastVol Volatility forecast to use in position size calculation as % 
32 // @returns Position size in units (shares, contracts, etc)
33 export volTargetedSize(float nominalBalance, float targetVolContribution, float forecastVol) =>
34     targetPos = nominalBalance / close * 0.01*targetVolContribution/forecastVol  // targetVolContribution input as %
35     if syminfo.type == "crypto"
36         targetPos := math.round(targetPos, 5)
37     else if syminfo.type == "forex"
38         targetPos := math.round(targetPos, 2)
39     else
40         targetPos := math.floor(targetPos)
41 
42     targetPos
43 
44 // example usage
45 balance = 10000
46 volEstimationWdw = 30
47 targetVolContribution = 15
48 volForecast = ewmaVol(volEstimationWdw)
49 targetPos = volTargetedSize(balance, targetVolContribution, volForecast)
50 plot(targetPos, title = "Target position", color = color.rgb(0, 120, 219))
