 4 //@version=5
 5 strategy("Vol-Targeted Position Size Strategy", overlay = false)
 6 import TradingView/ta/7
 7 import robotkris/VolTargeting/2 as VolTargeting
 8 
 9 balance = strategy.initial_capital
10 volEstimationWdw = input(30, "Volatility estimation window length")
11 targetVolContribution = input(20, "Target volatility contribution, %")
12 tradeBuffer = input(10, "Size of no-trade region as % of target position size")
13 
14 // calculate vol forecast as ewma of volatility
15 volForecast = VolTargeting.ewmaVol(volEstimationWdw)
16 
17 // calculate target position size
18 targetPos = VolTargeting.volTargetedSize(balance, targetVolContribution, volForecast)
19 upperBarrier = targetPos+targetPos*0.01*tradeBuffer
20 lowerBarrier = targetPos-targetPos*0.01*tradeBuffer
21 
22 // current position
23 float currentPosition = strategy.position_size
24 
25 // trade logic
26 delta = math.floor(math.abs(currentPosition - targetPos))
27 if currentPosition > upperBarrier
28     // current position is above trade buffer - rebalance back to target
29     strategy.order('position', direction = strategy.short, qty = delta)
30 else if currentPosition < lowerBarrier
31     // current position is below trade buffer - rebalance back to target
32     strategy.order('position', direction = strategy.long, qty = delta)
33 
34 // plot current position
35 plot(currentPosition, "current pos", color = color.rgb(37, 188, 0), style = plot.style_stepline, linewidth = 2)
36 // plot target position size and no-trade region
37 plot(targetPos, title = "Target position", color = color.rgb(66, 56, 245), style = plot.style_stepline)
38 barrier1 = plot(upperBarrier, "Top of no-trade region", color = color.rgb(245, 126, 56, 50))
39 barrier2 = plot(lowerBarrier, "Bottom of no-trade region", color = color.rgb(245, 126, 56, 50))
40 fill(barrier1, barrier2, color = color.rgb(245, 126, 56, 75))
